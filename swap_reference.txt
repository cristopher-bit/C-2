#include <iostream>
using namespace std;

void swapNumbers(int &a, int &b) {
    int temp = a;
    a = b;
    b = temp;
}

int main() {
    int x, y;
    cout << "Enter two numbers: ";
    cin >> x >> y;

    cout << "Before swapping: x = " << x << ", y = " << y << endl;

    swapNumbers(x, y);

    cout << "After swapping: x = " << x << ", y = " << y << endl;

    return 0;
}
Explanation
The function swapNumbers(int &a, int &b) takes two integers by reference.
Any modifications made inside the function directly affect the original variables.
No need to return values, as changes persist in the callerâ€™s scope.
